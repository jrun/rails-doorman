def _add_acl(type, args, block)
  opts = args.is_a?(Array) ? args.first : args
  if opts == :all
    @_doorman_default = type
    return true
  end
  if block
    _doorman_list << Rule.from_block(type, opts, &block)
  else
    _doorman_list << Rule.from_hash(type, opts)
  end
end
